import nl.knaw.huygens.hypercollate.rest.DotEngine
import java.io.ByteArrayOutputStream
import nl.knaw.huygens.hypercollate.model.CollationGraph
import jupyter.kotlin.*

fun CollationGraph.asSVG(): MimeTypedResult {
    val dot = this.asDot()
    val dotEngine = DotEngine("/usr/bin/dot")
    val outputStream = ByteArrayOutputStream()
    dotEngine.renderAs("svg",dot,outputStream)
    val svg = outputStream.toString("UTF-8")
    return MIME("image/svg+xml" to svg)
}
